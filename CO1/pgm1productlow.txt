\\1.Define a class ‘product’ with data members pcode, pname and price. Create 3 objects of
the class and find the product having the lowest price.

class Product{
    String pname;
    double price;
    int pcode; 

Product(int pcode , String pname , double price){
    this.pcode = pcode;
    this.pname = pname;
    this.price = price;
}
void display(){
    System.out.println(pcode+"\t\t"+pname+"\t\t"+price);
}
String lowPrice(Product p1 , Product p2 , Product p3){
    if(p1.price<p2.price && p1.price<p3.price){
            return p1.pname;
        }
        else if (p2.price<p1.price && p2.price<p3.price){
            return p2.pname;        }
        else {
            return p3.pname;        }
}
}

class PRD2{
    public static void main(String args[]){
        
        Product p1 = new Product(0456,"Cinnamon",1200);
        Product p2 = new Product(0542,"Cardamom",750);
        Product p3 = new Product(0105,"Saffron",1340);
        System.out.println("Product Information:\nProduct_Code\tProduct_Name\tProduct_Price");
	    p1.display();
	    p2.display();
	    p3.display();

        String low = p1.lowPrice(p1,p2,p3);
        System.out.println("Product having lowest price : " + low);       
        
    }
}
