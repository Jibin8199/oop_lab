//Write a program in Java to display the first 10 natural numbers.

public class Main {
  public static void main(String[] args) {
    int i=1;
    System.out.println("First 10 natural numbers are:");
    while (i<11) {
      System.out.println(i);
      i++;
    } 
  }
}
